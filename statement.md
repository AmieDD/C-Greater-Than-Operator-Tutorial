class GreaterThan
{
 static void Main()
    {

Console.WriteLine(0.0 > 2.1);   // output: False

Console.WriteLine(double.NaN > 2.1);   // output: False
Console.WriteLine(double.NaN <= 2.1);  // output: False
    }
}
